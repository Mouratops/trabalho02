✅ 1. Par ou Ímpar

HTML (ex1.html)

html
<input type="number" id="num">
<button onclick="verificarParImpar()">Verificar</button>
<p id="resultado"></p>

<script>
function verificarParImpar() {
  const num = parseInt(document.getElementById("num").value);
  const resultado = num % 2 === 0 ? "Par" : "Ímpar";
  document.getElementById("resultado").innerText = `O número é ${resultado}.`;
}
</script>

✅ 2. Tabuada de 1 a 10

html
<input type="number" id="numero">
<button onclick="mostrarTabuada()">Ver Tabuada</button>
<ul id="tabuada"></ul>

<script>
function mostrarTabuada() {
  const n = parseInt(document.getElementById("numero").value);
  let lista = "";
  for (let i = 1; i <= 10; i++) {
    lista += `<li>${n} x ${i} = ${n * i}</li>`;
  }
  document.getElementById("tabuada").innerHTML = lista;
}
</script>

✅ 3. Aumentar ou Diminuir Número

html
<p id="valor">0</p>
<button onclick="alterar(-1)">Diminuir</button>
<button onclick="alterar(1)">Aumentar</button>

<script>
let numero = 0;
function alterar(delta) {
  numero += delta;
  document.getElementById("valor").innerText = numero;
}
</script>

✅ 4. Adivinhação de Número

html
<input type="number" id="tentativa" min="0" max="10">
<button onclick="adivinhar()">Adivinhar</button>
<p id="resposta"></p>

<script>
const segredo = Math.floor(Math.random() * 11);
function adivinhar() {
  const chute = parseInt(document.getElementById("tentativa").value);
  const res = chute === segredo ? "Acertou!" : `Errou! O número era ${segredo}`;
  document.getElementById("resposta").innerText = res;
}
</script>

✅ 5. Média de 3 Notas

html
<input id="n1"><input id="n2"><input id="n3">
<button onclick="calcularMedia()">Calcular</button>
<p id="media"></p>

<script>
function calcularMedia() {
  const n1 = Number(document.getElementById("n1").value);
  const n2 = Number(document.getElementById("n2").value);
  const n3 = Number(document.getElementById("n3").value);
  const media = (n1 + n2 + n3) / 3;
  const msg = media >= 6 ? "Aprovado" : "Reprovado";
  document.getElementById("media").innerText = `Média: ${media.toFixed(2)} - ${msg}`;
}
</script>
